name: Environment-Specific Deployment
on:
  workflow_dispatch:
    inputs:
      environment:
        description: 'Select environment'
        required: true
        type: choice
        options:
          - uat
          - prod
  push:
    branches:
      - 'PIP-2135-ENV-TESTING'

jobs:
  deploy:
    strategy:
      matrix:
        config:
          - environment: uat
            runner-labels: ["self-hosted", "windows", "runners-us-west-2"]
            region: us-west-2
          - environment: prod
            runner-labels: ["self-hosted", "windows", "runners-us-east-1"]
            region: us-east-1
        include: ${{ fromJson(github.event_name == 'workflow_dispatch' && format('[{{"environment": "{0}"}}]', github.event.inputs.environment) || 
                  (contains(github.event.head_commit.message, '[env:uat]') && '[{"environment": "uat"}]' || 
                  (contains(github.event.head_commit.message, '[env:prod]') && '[{"environment": "prod"}]' || '[]') }}
    
    runs-on: ${{ matrix.config.runner-labels }}
    environment: ${{ matrix.config.environment == 'uat' && 'UAT' || 'Production' }}
    
    steps:
      - name: Checkout code
        uses: actions/checkout@v4
        
      - name: Show Environment Info
        run: |
          echo "Running on ${{ runner.name }}"
          echo "Environment: ${{ matrix.config.environment }}"
          echo "Region: ${{ matrix.config.region }}"
          echo "Trigger: ${{ github.event_name }}"
          echo "Commit Message: ${{ github.event.head_commit.message || 'N/A' }}"
          
      - name: Execute Deployment
        run: |
          echo "Deploying to ${{ matrix.config.environment }} environment"
          echo "Using region: ${{ matrix.config.region }}"
