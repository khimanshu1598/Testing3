runs:
  using: "composite"
  steps:
      - name: Get full path of PS1 script
        shell: powershell
        run: |
          $scriptPath = Resolve-Path "${{ github.action_path }}\..\..\od-gha-files\testscript.ps1"
          Write-Host "Full path to script: $scriptPath"
          & $scriptPath
        id: get-ps1-path
