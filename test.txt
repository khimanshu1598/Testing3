- name: Send sql-actions.ps1 to EC2
        uses: ./.github/actions/send-ps1-to-ec2
        with:
          script_path: "od-gha-files/sql-actions.ps1"
          instance_id: "i-0abc123def456ghij"
          region: "us-west-2"

      - name: Check if script exists in EmployeeInvitation folder
        shell: pwsh
        run: |
          aws ssm send-command `
            --document-name "AWS-RunPowerShellScript" `
            --instance-ids "i-0abc123def456ghij" `
            --region "us-west-2" `
            --parameters commands=["if (Test-Path 'C:\\Temp\\EmployeeInvitation\\sql-actions.ps1') { Write-Output '✅ File exists in EmployeeInvitation' } else { Write-Output '❌ File NOT found in EmployeeInvitation'; exit 1 }"] `
            --comment "Check for uploaded script in EmployeeInvitation"
